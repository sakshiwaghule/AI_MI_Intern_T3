import matplotlib.pyplot as plt
import seaborn as sns

# Set the visual style
sns.set(style="whitegrid")

# Define numerical columns
num_cols = df.select_dtypes(include=['float64', 'int64']).columns

# Create subplots
fig, axes = plt.subplots(nrows=2, ncols=2, figsize=(12, 10))
axes = axes.flatten()

for i, col in enumerate(num_cols):
    sns.histplot(df[col], kde=True, ax=axes[i], color='skyblue')
    axes[i].set_title(f'Distribution of {col}', fontsize=14)
    axes[i].set_xlabel(col)
    axes[i].set_ylabel('Frequency')

plt.tight_layout()
plt.savefig('numerical_distributions.png')

import pandas as pd
import numpy as np

# Load the dataset
df = pd.read_csv("C:/Users/waghu/OneDrive/Desktop/ELEVATE Labs/iris.csv")

# 1. Insights for Numerical Distributions (Histograms)
print("--- 1. Distribution Insights ---")
num_cols = df.select_dtypes(include=['float64', 'int64']).columns
for col in num_cols:
    mean = df[col].mean()
    std = df[col].std()
    print(f"{col}: Mean = {mean:.2f}, Std Dev = {std:.2f}")

# 2. Insights for Categorical Features (Count Plots)
print("\n--- 2. Categorical Analysis Insights ---")
species_counts = df['species'].value_counts()
print("Species counts:")
print(species_counts)
if species_counts.nunique() == 1:
    print("Insight: The dataset is perfectly balanced across all species.")

# 3. Insights for Outliers (Box Plots)
print("\n--- 3. Outlier Detection (using IQR method) ---")
for col in num_cols:
    Q1 = df[col].quantile(0.25)
    Q3 = df[col].quantile(0.75)
    IQR = Q3 - Q1
    lower_bound = Q1 - 1.5 * IQR
    upper_bound = Q3 + 1.5 * IQR
    outliers = df[(df[col] < lower_bound) | (df[col] > upper_bound)][col]
    if not outliers.empty:
        print(f"{col}: Found {len(outliers)} outliers: {outliers.values}")
    else:
        print(f"{col}: No outliers detected.")

# 4. Insights for Feature Relationships (Heatmap)
print("\n--- 4. Correlation Insights ---")
corr_matrix = df[num_cols].corr()
print("Correlation Matrix:")
print(corr_matrix)

# Highlighting strong correlations
high_corr = []
for i in range(len(corr_matrix.columns)):
    for j in range(i):
        if abs(corr_matrix.iloc[i, j]) > 0.8:
            high_corr.append((corr_matrix.columns[i], corr_matrix.columns[j], corr_matrix.iloc[i, j]))

print("\nStrong Correlations (>0.8 or <-0.8):")
for f1, f2, val in high_corr:
    print(f"{f1} and {f2}: {val:.2f}")

import pandas as pd
import numpy as np
from sklearn.ensemble import RandomForestClassifier
from sklearn.preprocessing import LabelEncoder

# Load the dataset
df = pd.read_csv("C:/Users/waghu/OneDrive/Desktop/ELEVATE Labs/iris.csv")
num_cols = df.select_dtypes(include=['float64', 'int64']).columns

# 1. Feature Importance calculation
X = df[num_cols]
y = df['species']
le = LabelEncoder()
y_enc = le.fit_transform(y)
rf = RandomForestClassifier(n_estimators=100, random_state=42)
rf.fit(X, y_enc)
importances = pd.Series(rf.feature_importances_, index=num_cols).sort_values(ascending=False)
top_feature = importances.index[0]

# 2. Dataset Balance check
counts = df['species'].value_counts()
is_balanced = len(counts.unique()) == 1

# 3. Outlier identification using IQR
outlier_info = []
for col in num_cols:
    Q1 = df[col].quantile(0.25)
    Q3 = df[col].quantile(0.75)
    IQR = Q3 - Q1
    outliers = df[(df[col] < (Q1 - 1.5 * IQR)) | (df[col] > (Q3 + 1.5 * IQR))]
    if len(outliers) > 0:
        outlier_info.append(f"{col} ({len(outliers)})")

# 4. Correlation Analysis
corr_matrix = df[num_cols].corr().abs()
# Get the pair with the highest correlation (excluding self-correlation)
sol = (corr_matrix.where(np.triu(np.ones(corr_matrix.shape), k=1).astype(bool))
                  .stack()
                  .sort_values(ascending=False))
top_pair = sol.index[0]
top_val = sol.iloc[0]

# Output findings
print("### Key Findings Summary:")
print(f"* **Class Balance**: The dataset is {'perfectly balanced' if is_balanced else 'unbalanced'}, containing {counts.iloc[0]} samples for each species.")
print(f"* **Predictive Power**: '{top_feature}' is the most critical feature for species classification, followed closely by petal width.")
print(f"* **Feature Redundancy**: There is a very strong correlation between **{top_pair[0]}** and **{top_pair[1]}** (${top_val:.2f}$), suggesting these features provide similar information.")
print(f"* **Outliers**: Outliers were detected in **{', '.join(outlier_info)}**. Other features are relatively clean.")
print(f"* **Separability**: Petal measurements exhibit bimodal distributions, which helps the model easily distinguish *Setosa* from the other species.")

### Key Findings Summary:
* **Class Balance**: The dataset is perfectly balanced, containing 50 samples for each species.
* **Predictive Power**: 'petal_length' is the most critical feature for species classification, followed closely by petal width.
* **Feature Redundancy**: There is a very strong correlation between **petal_length** and **petal_width** ($0.96$), suggesting these features provide similar information.
* **Outliers**: Outliers were detected in **sepal_width (4)**. Other features are relatively clean.
* **Separability**: Petal measurements exhibit bimodal distributions, which helps the model easily distinguish *Setosa* from the other species.
